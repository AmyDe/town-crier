```typescript
// web/src/api/client.ts

export class ApiRequestError extends Error {
  constructor(
    public readonly status: number,
    message: string,
  ) {
    super(message);
    this.name = 'ApiRequestError';
  }
}

export interface ApiClient {
  get<T>(path: string, params?: Record<string, string>): Promise<T>;
  post<T>(path: string, body?: unknown): Promise<T>;
  put(path: string, body?: unknown): Promise<void>;
  patch<T>(path: string, body?: unknown): Promise<T>;
  delete(path: string): Promise<void>;
}

export function createApiClient(
  baseUrl: string,
  getToken: () => Promise<string>,
  fetchFn: typeof globalThis.fetch = globalThis.fetch,
): ApiClient {
  async function request<T>(method: string, path: string, body?: unknown, params?: Record<string, string>): Promise<T> {
    const token = await getToken();
    let url = `${baseUrl}${path}`;
    if (params) {
      const search = new URLSearchParams(params);
      url = `${url}?${search.toString()}`;
    }

    const headers: Record<string, string> = {
      Authorization: `Bearer ${token}`,
    };
    if (body !== undefined) {
      headers['Content-Type'] = 'application/json';
    }

    const response = await fetchFn(url, {
      method,
      headers,
      body: body !== undefined ? JSON.stringify(body) : undefined,
    });

    if (!response.ok) {
      let message = `Request failed with status ${response.status}`;
      try {
        const errorBody = await response.json() as { error?: string };
        if (errorBody.error) {
          message = errorBody.error;
        }
      } catch {
        // Use default message if body isn't JSON
      }
      throw new ApiRequestError(response.status, message);
    }

    if (response.status === 204) {
      return undefined as T;
    }

    return response.json() as Promise<T>;
  }

  return {
    get: <T>(path: string, params?: Record<string, string>) => request<T>('GET', path, undefined, params),
    post: <T>(path: string, body?: unknown) => request<T>('POST', path, body),
    put: (path: string, body?: unknown) => request<void>('PUT', path, body),
    patch: <T>(path: string, body?: unknown) => request<T>('PATCH', path, body),
    delete: (path: string) => request<void>('DELETE', path),
  };
}
```

```typescript
// web/src/api/useApiClient.ts
import { createContext, useContext } from 'react';
import type { ApiClient } from './client';

export const ApiClientContext = createContext<ApiClient | null>(null);

export function useApiClient(): ApiClient {
  const client = useContext(ApiClientContext);
  if (!client) {
    throw new Error('useApiClient must be used within an ApiClientProvider');
  }
  return client;
}
```

```tsx
// web/src/api/ApiClientProvider.tsx
import type { ReactNode } from 'react';
import { useMemo } from 'react';
import { useAuth0 } from '@auth0/auth0-react';
import { createApiClient } from './client';
import { ApiClientContext } from './useApiClient';

const API_BASE_URL = import.meta.env.VITE_API_BASE_URL as string || 'http://localhost:5000';

interface Props {
  children: ReactNode;
}

export function ApiClientProvider({ children }: Props) {
  const { getAccessTokenSilently } = useAuth0();

  const client = useMemo(
    () => createApiClient(API_BASE_URL, getAccessTokenSilently),
    [getAccessTokenSilently],
  );

  return (
    <ApiClientContext.Provider value={client}>
      {children}
    </ApiClientContext.Provider>
  );
}
```

```tsx
import { useEffect } from 'react';
import { useAuth0 } from '@auth0/auth0-react';
import { Outlet } from 'react-router-dom';

export function AuthGuard() {
  const { isAuthenticated, isLoading, loginWithRedirect } = useAuth0();

  useEffect(() => {
    if (!isLoading && !isAuthenticated) {
      loginWithRedirect();
    }
  }, [isLoading, isAuthenticated, loginWithRedirect]);

  if (isLoading || !isAuthenticated) {
    return null;
  }

  return <Outlet />;
}
```

```tsx
// web/src/auth/OnboardingGate.tsx
import { useEffect, useState } from 'react';
import { Outlet, useNavigate } from 'react-router-dom';
import { useApiClient } from '../api/useApiClient';
import { ApiRequestError } from '../api/client';
import type { UserProfile } from '../domain/types';

export function OnboardingGate() {
  const client = useApiClient();
  const navigate = useNavigate();
  const [checked, setChecked] = useState(false);

  useEffect(() => {
    let cancelled = false;
    client.get<UserProfile>('/v1/me').then(() => {
      if (!cancelled) setChecked(true);
    }).catch((err: unknown) => {
      if (!cancelled) {
        if (err instanceof ApiRequestError && err.status === 404) {
          navigate('/onboarding', { replace: true });
        } else {
          setChecked(true);
        }
      }
    });
    return () => { cancelled = true; };
  }, [client, navigate]);

  if (!checked) return null;

  return <Outlet />;
}
```

```tsx
// web/src/shared/AuthoritySelector/AuthoritySelector.tsx
import { useState, useEffect, useRef } from 'react';
import type { AuthorityListItem } from '../../domain/types';
import styles from './AuthoritySelector.module.css';

interface Props {
  onSearch: (query: string) => Promise<AuthorityListItem[]>;
  onSelect: (authority: AuthorityListItem) => void;
  placeholder?: string;
}

export function AuthoritySelector({ onSearch, onSelect, placeholder = 'Search authorities...' }: Props) {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState<AuthorityListItem[]>([]);
  const [showDropdown, setShowDropdown] = useState(false);
  const [searched, setSearched] = useState(false);
  const debounceRef = useRef<ReturnType<typeof setTimeout>>();

  useEffect(() => {
    if (query.length < 2) {
      setResults([]);
      setShowDropdown(false);
      setSearched(false);
      return;
    }

    clearTimeout(debounceRef.current);
    debounceRef.current = setTimeout(async () => {
      const items = await onSearch(query);
      setResults(items);
      setShowDropdown(true);
      setSearched(true);
    }, 250);

    return () => clearTimeout(debounceRef.current);
  }, [query, onSearch]);

  function handleSelect(authority: AuthorityListItem) {
    setQuery(authority.name);
    setShowDropdown(false);
    onSelect(authority);
  }

  return (
    <div className={styles.container}>
      <input
        type="text"
        className={styles.input}
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder={placeholder}
      />
      {showDropdown && (
```

```tsx
// web/src/features/Onboarding/OnboardingPage.tsx
import { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { useApiClient } from '../../api/useApiClient';
import { PostcodeInput } from '../../shared/PostcodeInput/PostcodeInput';
import { RadiusPicker } from '../../shared/RadiusPicker/RadiusPicker';
import type { Coordinates } from '../../domain/types';
import styles from './OnboardingPage.module.css';

type Step = 'welcome' | 'postcode' | 'radius';

const FREE_RADIUS_OPTIONS = [500, 1000, 2000];

export function OnboardingPage() {
  const client = useApiClient();
  const navigate = useNavigate();
  const [step, setStep] = useState<Step>('welcome');
  const [postcode, setPostcode] = useState('');
  const [coords, setCoords] = useState<Coordinates | null>(null);
  const [radius, setRadius] = useState(1000);
  const [saving, setSaving] = useState(false);

  async function handleGeocode(pc: string) {
    const result = await client.get<{ coordinates: Coordinates }>(`/v1/geocode/${encodeURIComponent(pc)}`);
    setPostcode(pc);
    setCoords(result.coordinates);
    setStep('radius');
  }

  async function handleFinish() {
    if (!coords) return;
    setSaving(true);
    try {
      await client.post('/v1/me');
      await client.post('/v1/me/watch-zones', {
        zoneId: crypto.randomUUID(),
        name: postcode,
        latitude: coords.latitude,
        longitude: coords.longitude,
        radiusMetres: radius,
      });
      navigate('/dashboard', { replace: true });
    } finally {
      setSaving(false);
    }
  }

  return (
    <div className={styles.container}>
      {step === 'welcome' && (
        <div className={styles.step}>
          <h1>Welcome to Town Crier</h1>
          <p className={styles.subtitle}>
            Monitor planning applications in your neighbourhood. Let&apos;s set up your first watch zone.
          </p>
          <button className={styles.primaryButton} onClick={() => setStep('postcode')}>
            Get started
          </button>
        </div>
      )}

      {step === 'postcode' && (
        <div className={styles.step}>
          <h2>Enter your postcode</h2>
          <p className={styles.subtitle}>We&apos;ll use this to find planning applications near you.</p>
          <PostcodeInput onGeocode={handleGeocode} />
        </div>
      )}

      {step === 'radius' && (
        <div className={styles.step}>
          <h2>Choose your radius</h2>
          <p className={styles.subtitle}>How far around {postcode} should we monitor?</p>
          <RadiusPicker options={FREE_RADIUS_OPTIONS} value={radius} onChange={setRadius} />
          <button
            className={styles.primaryButton}
            onClick={handleFinish}
            disabled={saving}
            aria-label="Finish"
          >
            {saving ? 'Setting up...' : 'Finish'}
          </button>
        </div>
      )}
    </div>
  );
}
```
